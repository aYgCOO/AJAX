jQuery Syntax and Concepts

Basic jQuery Syntax and Concepts

1. Including jQuery

To use jQuery, include it in your HTML file. You can do this by linking to a CDN or downloading it and including it locally.

<!-- CDN -->
<script src="https://ajax.googleapis.com/ajax/libs/jquery/3.5.1/jquery.min.js"></script>
<script src="https://code.jquery.com/ui/1.12.1/jquery-ui.min.js"></script>

2. Document Ready Event

Ensures that the DOM is fully loaded before executing any jQuery code.

$(document).ready(function(){
    // Your code here
});

3. Selecting Elements

Use the $() function to select elements.

$("p").hide(); // Hides all <p> elements

4. CSS Manipulation

Change the CSS properties of elements.

$("body").css("font-family", "'Segoe UI'");
$("#main").css({ padding: "20px", margin: "auto", width: "80vw"});
$("a").css({ fontWeight: "bolder", textDecoration: "none" });

5. jQuery UI Components

jQuery UI provides a set of user interface interactions, effects, widgets, and themes built on top of jQuery.

Accordion

$("#accordion").accordion();

Tabs

$("#tabs").tabs();

Autocomplete

let arr = ["ActionScript", "AppleScript", "Asp", "BASIC", "C", "C++", "Clojure", "COBOL", "ColdFusion", "Erlang", "Fortran", "Groovy", "Haskell", "Java", "JavaScript", "Lisp", "Perl", "PHP", "Python", "Ruby", "Scala", "Scheme"];
$("#autocomplete").autocomplete({
    source: arr,
});

Button

$("#btn").button({
    classes: {
        "ui-button": "highlight"
    }
});

Draggable

$("#draggable").draggable();

Droppable

$("#droppable").droppable({
    drop: function (event, ui) {
        $(this).addClass("ui-state-highlight").find("p").html("Dropped!");
    }
});

Resizable

$("#resizable").resizable();

Dialog

$("#dialog").dialog();

Tooltip

$("#tooltip").tooltip();

Effects

$("#react").button().click(function () {
    $("#effect").toggle("blind", {}, 500);
});

Open Dialog

$("#pop").dialog({
    modal: true,
    autoOpen: false,
    buttons: {
        "Ok": function(){
            $(this).dialog("close");
        },
        "Close": function(){
            $(this).dialog("close");
        }
    }
});
$("#open-dialog").button().on("click", function(){
    $("#pop").dialog("open");
});
